يجب تفعيل صلاحيات جدول stories

اذهب إلى Supabase → SQL Editor والصق هذا الكود:

-- 1) جدول الستوريات CREATE TABLE IF NOT EXISTS stories ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, user_id UUID, image_url TEXT NOT NULL, text_overlay TEXT DEFAULT '', font_style TEXT DEFAULT 'Tajawal', text_color TEXT DEFAULT '#ffffff', created_at TIMESTAMPTZ DEFAULT NOW(), expires_at TIMESTAMPTZ DEFAULT (NOW() + INTERVAL '24 hours') ); ALTER TABLE stories DISABLE ROW LEVEL SECURITY; -- 2) جدول المشاهدات CREATE TABLE IF NOT EXISTS story_views ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, story_id UUID REFERENCES stories(id) ON DELETE CASCADE, viewer_id UUID, created_at TIMESTAMPTZ DEFAULT NOW(), UNIQUE(story_id, viewer_id) ); CREATE TABLE IF NOT EXISTS story_likes ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, story_id UUID REFERENCES stories(id) ON DELETE CASCADE, viewer_id UUID, created_at TIMESTAMPTZ DEFAULT NOW(), UNIQUE(story_id, viewer_id) ); ALTER TABLE story_views DISABLE ROW LEVEL SECURITY; ALTER TABLE story_likes DISABLE ROW LEVEL SECURITY;
